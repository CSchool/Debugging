Юному хакеру Васе удалось взломать школьный сервер и скачать оттуда программу, проверяющую пароль при обращении к этому серверу. Ниже приведена эта программа на двух языках - C++ и Паскаль. Она запрашивает с клавиатуры пароль пользователя. Если пароль правильный, программа выдает сообщение "Access granted!".
C++Паскаль
#include <stdio.h>
#include <string.h>

int main()
{
    char pass[256];

    printf("Enter password: ");
    gets(pass);

    int sum = 0; 
    for (char *p = pass; *p ; ++p)
    {
        sum += *p - 'a';
    }
    if (sum % 124 == 18 && strlen(pass) > 10)
        printf("Access granted!\n");
    else
        printf("Access denied.\n");
    return 0;
}

var
    pass: string;
    sum: integer;
    i: integer;

begin
    write('Enter password: ');
    readln(pass);

    sum:=0;
    for i := 1 to length(pass) do
        inc(sum, ord(pass[i]) - ord('a'));

    if (sum mod 124 = 18) and (length(pass) > 10) then
        writeln('Access granted!')
    else
        writeln('Access denied.');
end.
