Оказалось, что первая программа, найденная Васей на сервере, была просто задачей для олимпиады. Тогда Вася продолжил и поиски и наткнулся на другую программу. Она запрашивает с клавиатуры имя и пароль пользователя. Если пароль правильный, программа выдает сообщение "Access granted!".
C++Паскаль
#include <stdio.h>
#include <string.h>

int main()
{
    char buf[256];
    char pass[256];

    printf("Enter your name: ");
    gets(buf);
    int sum1 = 0;
    int i = 0;

    for (char *p = buf ; *p ; ++p, i++)
    {
        if (i % 5 == 0)
            sum1 += (*p - '0');
    }

    printf("Enter password: ");
    gets(pass);

    int sum2 = 0; 
    for (char *p = pass; *p ; ++p)
    {
        sum2 += (*p - '0') * (*p - '0') * (*p - '0');
    }
    if ((sum1 + sum2) % 21 == 9)
        printf("Access granted!\n");
    else
        printf("Access denied.\n");
    return 0;
}

var
    buf, pass: string;
    sum1, sum2: longint;
    i: integer;

begin
    write('Enter your name: ');
    readln(buf);

    for i := 1 to length(buf) do begin
        if ((i - 1) mod 5 = 0) then
           inc(sum1, ord(buf[i]) - ord('0'));
    end;

    write('Enter password: ');
    readln(pass);

    for i := 1 to length(pass) do
        inc(sum2, (ord(pass[i]) - ord('0'))
                  * (ord(pass[i]) - ord('0'))
                  * (ord(pass[i]) - ord('0')));

    if ((sum1 + sum2) mod 21 = 9) then
        writeln('Access granted!')
    else
        writeln('Access denied.');
end.


Теперь Вася хочет узнать пароль администратора, чтобы исправить свои годовые оценки.
Напишите программу, восстанавливающую пароль по заданному имени.

Входные данные
Одна строка с именем пользователя. Длина строки не превышает 255 символов.

Выходные данные
Выведите одну строку - пароль, который примет программа из условия.
